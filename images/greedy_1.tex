\node[text width=100mm] {
	\begin{pseudocode}
		\myproc{LargestConcatenate}(\myvar{Numbers}):\\
		$\myvar{result} \gets \text{empty string}$\\
		while \myvar{Numbers} is not empty:\\
		\quad $\myvar{maxNumber} \gets \text{largest among \myvar{Numbers}}$\\
		\quad append \myvar{maxNumber} to \myvar{result}\\
		\quad remove \myvar{maxNumber} from \myvar{Numbers}\\
		return \myvar{result}
	\end{pseudocode}
};
