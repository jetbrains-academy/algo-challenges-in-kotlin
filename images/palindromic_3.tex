\node[text width=100mm] {
	\begin{pseudocode}
		$\myproc{LPS}(s)$:\\
		$n \gets \text{length of $s$}$\\
		$\myvar{table} \gets \text{$(n+1) \times (n+1)$ array}$
		
		for $i$ from $0$ to $n$:\\
		\quad $\myvar{table}[i,i] \gets 0$\\
		\quad if $i < n$:\\
		\quad\quad $\myvar{table}[i,i+1] \gets 1$\\
		
		for $i$ from $0$ to $n$:\\
		\quad for $\myvar{size}$ from $2$ to $n-i$:\\
		\quad\quad $j \gets i + \myvar{size}$\\
		\quad\quad if $s[i]=s[j-1]$:\\
		\quad\quad\quad $\myvar{table}[i,j] \gets \myvar{table}[i+1, j-1]+2$\\
		\quad\quad else:\\
		\quad\quad\quad $\myvar{table}[i,j] \gets \max(\myvar{table}[i+1, j], \myvar{table}[i, j-1])$\\
		
		return $\myvar{table}[0,n]$
	\end{pseudocode}
};
