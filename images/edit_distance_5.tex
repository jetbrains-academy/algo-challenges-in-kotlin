\node[text width=100mm]{
	\hrule
	\medskip
	$\myproc{EditDistance}(A[1 \dotsc n], B[1 \dotsc m]):$\\
	$\myvar{table} \gets \text{2d array of size $(n + 1) \times (m + 1)$}$\\
	$\myvar{table}[i][0] \gets i$ and $\myvar{table}[0][j] \gets j$ for all $i,j$\\
	for $i$ from~$1$ to~$n$:\\
	\quad for $j$ from~$1$ to~$m$:\\
	\quad\quad \textcolor{two}{$\myvar{insertion} \gets \myvar{table}[i][j-1]+1$}\\
	\quad\quad \textcolor{one}{$\myvar{deletion} \gets \myvar{table}[i-1][j]+1$}\\
	\quad\quad \textcolor{three}{$\myvar{match} \gets \myvar{table}[i-1][j-1]$}\\
	\quad\quad \textcolor{four}{$\myvar{mismatch} \gets \myvar{table}[i-1][j-1]+1$}\\
	\quad\quad if $A[i]=B[j]$:\\
	\quad\quad\quad $\myvar{table}[i][j] \gets \min(\textcolor{two}{\myvar{insertion}}, \textcolor{one}{\myvar{deletion}}, \textcolor{three}{\myvar{match}})$\\
	\quad\quad else:\\
	\quad\quad\quad $\myvar{table}[i][j]\gets \min(\textcolor{two}{\myvar{insertion}}, \textcolor{one}{\myvar{deletion}}, \textcolor{four}{\myvar{mismatch}})$\\
	return $\myvar{table}[n][m]$
	\medskip
	\hrule
};
