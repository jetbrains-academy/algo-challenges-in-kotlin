\node[text width=100mm] {
	\begin{pseudocode}
    	$\myvar{table} \gets \text{associative array}$\\

		\myproc{Change}(\myvar{money}):\\
		if $\myvar{table}[\myvar{money}]$ is not yet computed:\\
		\quad if $\myvar{money}=0$:\\
		\quad\quad $\myvar{table}[\myvar{money}] \gets 0$\\
		\quad else:\\
		\quad\quad$\myvar{result} \gets +\infty$\\
		\quad\quad for $c=1,3,4$:\\
		\quad\quad\quad if $c \le \myvar{money}$:\\
		\quad\quad\quad\quad $\myvar{result} \gets \min(\myvar{result},  1+\myproc{Change}(\myvar{money}-c))$\\
		\quad\quad $\myvar{table}[\myvar{money}] \gets \myvar{result}$\\
		return $\myvar{table}[\myvar{money}]$
	\end{pseudocode}
};
