\node[text width=100mm] {
	\begin{pseudocode}
		\myproc{Change}(\myvar{money}):\\
		if $\myvar{money}=0$:\\
		\quad return~$0$\\
		else:\\
		\quad$\myvar{result} \gets +\infty$\\
		\quad for $c=1,3,4$:\\
		\quad\quad if $c \le \myvar{money}$:\\
		\quad\quad\quad $\myvar{result} \gets \min(\myvar{result},  1+\myproc{Change}(\myvar{money}-c))$\\
		\quad return \myvar{result}
	\end{pseudocode}
};
