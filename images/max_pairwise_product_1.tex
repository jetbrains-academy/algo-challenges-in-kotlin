\node[text width=100mm]{
	\begin{pseudocode}
		\myproc{MaxPairwiseProductNaive}($A[1\dotsc n]$):\\
		$\myvar{product} \gets 0$\\
		for $i$ from $1$ to~$n$:\\
		\quad for $j$ from $1$ to~$n$:\\
		\quad\quad if $i \neq j$:\\
		\quad\quad\quad if $\myvar{product} < A[i] \cdot A[j]$:\\
		\quad\quad\quad\quad $\myvar{product} \gets A[i] \cdot A[j]$\\
		return \myvar{product}
	\end{pseudocode}
};
